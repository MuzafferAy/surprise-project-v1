    <script>
        const noBtn = document.getElementById('noBtn');
        const yesBtn = document.getElementById('yesBtn');
        const question = document.getElementById('question');
        let yesScale = 1;

        const moveNo = (e) => {
            if(e) e.preventDefault();
            const padding = 100;
            const maxX = window.innerWidth - noBtn.offsetWidth - padding;
            const maxY = window.innerHeight - noBtn.offsetHeight - padding;
            
            const randomX = Math.max(padding, Math.floor(Math.random() * maxX));
            const randomY = Math.max(padding, Math.floor(Math.random() * maxY));
            
            noBtn.style.left = randomX + 'px';
            noBtn.style.top = randomY + 'px';

            yesScale += 0.35; // Her hayırda Evet daha çok büyür
            yesBtn.style.transform = `scale(${yesScale})`;
            
            if(yesScale > 5) { // Artık kaçacak yer kalmadığında
                noBtn.style.display = 'none';
            }
        };

        noBtn.addEventListener('touchstart', moveNo);
        noBtn.addEventListener('mouseover', moveNo);

        yesBtn.addEventListener('click', () => {
            document.body.innerHTML = `
                <div style="display:flex; flex-direction:column; align-items:center; justify-content:center; height:100vh; background:#fff;">
                    <h1 style="font-size:3.5rem; color:#ff4d6d; text-align:center; animation: pulse 1s infinite;">Seni Çok Seviyorum! ❤️<br><span style="font-size:1.5rem">İyi ki varsın...</span></h1>
                </div>
            `;
            
            // Konfeti Yağmuru
            var duration = 15 * 1000;
            var animationEnd = Date.now() + duration;
            var defaults = { startVelocity: 30, spread: 360, ticks: 60, zIndex: 0 };

            function randomInRange(min, max) { return Math.random() * (max - min) + min; }

            var interval = setInterval(function() {
                var timeLeft = animationEnd - Date.now();
                if (timeLeft <= 0) return clearInterval(interval);
                var particleCount = 50 * (timeLeft / duration);
                confetti(Object.assign({}, defaults, { particleCount, origin: { x: randomInRange(0.1, 0.3), y: Math.random() - 0.2 } }));
                confetti(Object.assign({}, defaults, { particleCount, origin: { x: randomInRange(0.7, 0.9), y: Math.random() - 0.2 } }));
            }, 250);
        });
    </script>
